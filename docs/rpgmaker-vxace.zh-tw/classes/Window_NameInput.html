def item_rect(index)
  rect = Rect.new
  rect.x = index % 10 * 32 + index % 10 / 5 * 16
  rect.y = index / 10 * line_height
  rect.width = 32
  rect.height = line_height
  rect
end

def on_name_ok
  if @edit_window.name.empty?
    if @edit_window.restore_default
      Sound.play_ok
    else
      Sound.play_buzzer

  else
    Sound.play_ok

  end
end

# File Window_NameInput.rb, line 215
def process_ok
  if !character.empty?
    on_name_add
  elsif is_page_change?
    Sound.play_ok
    cursor_pagedown
  elsif is_ok?
    on_name_ok
  end
end
