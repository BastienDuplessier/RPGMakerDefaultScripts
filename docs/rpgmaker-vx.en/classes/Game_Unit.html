def average_agi
  result = 0
  n = 0
  for member in members
    result += member.agi
    n += 1
  end
  result /= n if n > 0
  result = 1 if result == 0
  return result
end

def random_target
  roulette = []
  for member in existing_members
    member.odds.times do
      roulette.push(member)
    end
  end
  return roulette.size > 0 ? roulette[rand(roulette.size)] : nil
end
