def include?(item)
  return true if item == nil
  if @equip_type == 0
    return false unless item.is_a?(RPG::Weapon)
  else
    return false unless item.is_a?(RPG::Armor)
    return false unless item.kind == @equip_type - 1
  end
  return @actor.equippable?(item)
end
