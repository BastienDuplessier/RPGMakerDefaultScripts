def update
  # 刷新点灭记数
  @blink_count = (@blink_count + 1) % 8
  # 设置传送源矩形
  if @blink_count < 4
    self.src_rect.set(128, 96, 32, 32)
  else
    self.src_rect.set(160, 96, 32, 32)
  end
  # 刷新帮助文本 (update_help 定义了继承目标)
  if @help_window != nil
    update_help
  end
end
