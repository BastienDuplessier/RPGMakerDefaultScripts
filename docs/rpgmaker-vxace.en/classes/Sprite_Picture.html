def update_bitmap
  if @picture.name.empty?
    self.bitmap = nil
  else
    self.bitmap = Cache.picture(@picture.name)
  end
end

def update_origin
  if @picture.origin == 0
    self.ox = 0
    self.oy = 0
  else
    self.ox = bitmap.width / 2
    self.oy = bitmap.height / 2
  end
end
