# File Game_System.rb, line 45
def bgm_play(bgm)
  @playing_bgm = bgm
  if bgm != nil and bgm.name != ""
    Audio.bgm_play("Audio/BGM/" + bgm.name, bgm.volume, bgm.pitch)
  else
    Audio.bgm_stop
  end
  Graphics.frame_reset
end
