def draw_line(index)
  rect = Rect.new(0, 0, 0, 0)
  rect.x += 4
  rect.y += index * WLH
  rect.width = contents.width - 8
  rect.height = WLH
  self.contents.clear_rect(rect)
  self.contents.font.color = normal_color
  self.contents.draw_text(rect, @lines[index])
end
