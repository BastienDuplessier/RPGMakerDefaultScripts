def update
  if @timer_working and @timer > 0
    @timer -= 1
    if @timer == 0 and $game_temp.in_battle     # 戦闘中にタイマーが 0 に
      $game_temp.next_scene = "map"             # なったら戦闘を中断する
    end
  end
end
